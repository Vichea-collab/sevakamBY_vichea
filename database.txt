Sevakam Firestore Database Schema

Users Collection
Path: users/{userId}
- name: String
- email: String
- role: String (e.g., 'finder', 'provider', 'admin')
- photoUrl: String
- createdAt: Timestamp

Sub-collections:
Notifications
Path: users/{userId}/notification/{notificationId}
- type: String [accepted, rejected, incoming order, announcement]
- title: String
- body: String
- refType: String
- refId: String (e.g., bookingId)
- isRead: Boolean
- createdAt: Timestamp

Settings
Path: users/{userId}/settings/notification
- pushEnabled: Boolean
- orderEnabled: Boolean
- chatEnabled: Boolean
- announcementEnabled: Boolean

FCM Tokens
Path: users/{userId}/fcmTokens/{tokenId}
- token: String
- platform: String [android, ios]
- lastSeenAt: Timestamp
- createdAt: Timestamp

---

Provider Collection
Path: provider/{userId}
- bio: String
- photoUrl: String
- phoneNumber: String
- location: GeoPoint
- city: String
- ratePerHour: Number
- serviceId: String
- serviceName: String
- serviceImage: String
- birthday: Timestamp
- ratingCount: Number
- ratingSum: Number
- activeOrder: Number
- completedOrder: Number
- createdAt: Timestamp

Sub-collections:
Wallet
Path: provider/{userId}/wallet/main
- balance: Number
- createdAt: Timestamp

Transactions
Path: provider/{userId}/wallet/main/transaction/{transactionId}
- amount: Number
- type: String [deposit, pay commission]
- reference: String (for deposits)
- bookingId: String (for commission)
- note: String
- createdAt: Timestamp

Finder Collection

- photoUrl: String
- phoneNumber: String
- birthday: Timestamp
- location: GeoPoint
- city: String

Service Collection

- serviceName: String
- description: String
- categoryId: String
- serviceImage: String
- bookingCount: Number
- completedCount: Number

Category Collection

Review Collection

- comment: String
- rating: Number

- finderName: String
- finderProfile: String

- bookingId: String

Booking Collection
Path: booking/{bookingId}
- finderId: String
- providerId: String
- serviceId: String
- homeType: String
- ratePerHour: Number
- location: GeoPoint
- city: String
- bookingStatus: String (e.g., pending, accepted, completed, cancelled)
- paymentStatus: String (e.g., pending, paid)
- startDate: Timestamp
- updatedAt: Timestamp
- createdAt: Timestamp

Provider Posts Collection
Path: providerPost/{providerPostId}
- providerId: String
- providerName: String
- providerPfPicture: String
- caption: String
- pictureUrl: String
- isPublic: Boolean
- createdAt: Timestamp

---

Finder Request Posts Collection
Path: finderRequestPost/{finderRequestPostId}
- finderId: String
- finderName: String
- finderPfPicture: String
- serviceId: String
- caption: String
- image: List<String> (Max: 3)
- location: GeoPoint
- city: String
- status: String [open, assigned, cancelled]
- bookingId: String (Nullable)
- assignedToProviderId: String (Nullable)
- createdAt: Timestamp

Ads Collection

- adsImage: String
- expiryDate: Timestamp
- isActive: Boolean
- createdAt: Timestamp

Conversation Collection
Path: conversation/{conversationId}
- finderId: String
- providerId: String
- lastMessage: String
- lastMessageAt: Timestamp
- lastSenderId: String
- finderUnreadCount: Number
- providerUnreadCount: Number
- isBlocked: Boolean
- createdAt: Timestamp

Sub-collections:
Messages
Path: conversation/{conversationId}/message/{messageId}
- senderId: String
- type: String [text, image]
- text: String
- imageUrl: String
- createdAt: Timestamp
